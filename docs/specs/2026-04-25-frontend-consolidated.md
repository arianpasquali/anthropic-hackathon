### Bank schema

```typescript
interface Bank {
  id: string                        // "rotterdam" | "amsterdam" | "haaglanden" | "eindhoven" | "groningen" | "breda"
  name: string
  region: string                    // e.g. "Zuid-Holland"
  annual_kg_rescued: number
  annual_tco2e: number
  weighted_emission_factor: number
  households_weekly: number
  people_served: number | null
  is_rdc: boolean                   // is regional distribution center
  rdc_satellite_count: number | null
  cluster_banks: string[]
  category_mix: CategoryMix
  source_url: string
  provenance: ProvenanceRecord[]
  standard_package: StandardPackage | null
  // to add:
  lat: number
  lng: number
}

interface CategoryMix {
  produce: number      // fraction 0–1, sum = 1
  dry_goods: number
  dairy: number
  bakery: number
  meat: number
  prepared: number
  eggs: number
}

interface StandardPackage {
  amount_eur: number               // 25000
  attributed_tco2e: number         // kg CO₂e attributed to this package
  attributed_kg_food: number
  attribution_share: number        // fraction of bank's annual baseline
  is_solo_capable: boolean
  category_breakdown: {
    [category: string]: {
      tco2e: number
      kg_food: number
    }
  }
}
```

### 6. Methodology (`/methodology`)

**Existing route — full redesign as animated infographic.**

Sections (scroll-triggered reveals via IntersectionObserver):

1. **Hero** — dark emerald gradient, "How we calculate avoided CO₂e", badges: FRAME Aligned · CSRD-Ready · ESRS E1+S3 · NL-specific


7. **Audit trail** — 6 numbered steps from submission → CSR report
8. **Trust pillars** — 3 cards: deterministic formula · source citations · conservative NL baseline

## `src/lib/banks.ts` Extensions

Add to existing `banks.ts` (do not rewrite):

```typescript
// Coordinates per bank (supplement banks.json or hardcode here)
export const BANK_COORDS: Record<string, { lat: number; lng: number }> = {
  rotterdam:  { lat: 51.9225, lng: 4.4792 },
  amsterdam:  { lat: 52.3676, lng: 4.9041 },
  haaglanden: { lat: 52.0705, lng: 4.3007 },
  eindhoven:  { lat: 51.4416, lng: 5.4697 },
  groningen:  { lat: 53.2194, lng: 6.5665 },
  breda:      { lat: 51.5719, lng: 4.7683 },
}

// Compute CO₂e per category for a bank
export function computeCategoryTco2e(bank: Bank): Record<string, number> {
  return Object.entries(bank.category_mix).reduce((acc, [cat, frac]) => {
    acc[cat] = bank.annual_kg_rescued * frac * EMISSION_FACTORS[cat] * CF_NL
    return acc
  }, {} as Record<string, number>)
}
```

## Data Types (TypeScript — for backend contract alignment)

```typescript
// Mirrors Python enums.py
type ImpactProfileEnum = 'co2_focus' | 'social_focus' | 'balanced'
type SourceEnum = 'extracted' | 'inferred_national_avg' | 'inferred_category_split' | 'inferred_calculation' | 'manual'
type StatusEnum = 'pending' | 'paid' | 'failed' | 'refunded'
type TemplateEnum = 'gri' | 'csrd' | 'esrs_e1' | 'generic'

// Mirrors Python marketplace.py
interface FundSubscription {
  id: string
  user_id: string
  package_id: string        // maps to bank id in current data model
  amount_eur: number
  status: StatusEnum
  solvimon_id?: string
  purchased_at: string      // ISO-8601
  csr_report_id?: string
}

// Mirrors Python frame.py FrameResult
interface FrameResult {
  id: string
  report_id: string
  co2e_total_kg: number
  co2e_produce_kg: number
  co2e_meat_fish_kg: number
  co2e_dairy_eggs_kg: number
  co2e_dry_goods_kg: number
  co2e_bread_kg: number
  emission_factor_source: string
  methodology_version: string
  computed_at: string       // ISO-8601
}

// Mirrors Python marketplace.py CsrReport
interface CsrReport {
  id: string
  subscription_id: string
  frame_result_id: string
  generated_at: string      // ISO-8601
  file_path: string
  template: TemplateEnum
}
```
